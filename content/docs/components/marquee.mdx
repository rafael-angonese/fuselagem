### Update `tailwind.config.js`

Add the following animations to your `tailwind.config.js` file.

```js {5-12}
/** @type {import('tailwindcss').Config} */
module.exports = {
  theme: {
    extend: {
      animation: {
        marquee: "marquee var(--duration, 30s) linear infinite",
      },
      keyframes: {
         marquee: {
          to: { transform: 'translateX(-50%)' }
        },
      },
    },
  },
};
```
