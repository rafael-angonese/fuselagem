### Update `tailwind.config.js`

Add the following animations to your `tailwind.config.js` file.

```js {5-17}
/** @type {import('tailwindcss').Config} */
module.exports = {
  theme: {
    extend: {
      keyframes: {
        gradient: {
          to: {
            backgroundPosition: "var(--bg-size) 0",
          },
        },
      },
      animation: {
        gradient: "gradient 8s linear infinite",
      },
    },
  },
};
```
